        <script>
            let map;
            let marker; // Declarar una variable para el marcador
            let direccionInput = document.getElementById('direccionInput'); // Obtener el campo de texto de dirección

            // Inicializa el mapa
            function initMap() {
                // Coordenadas iniciales en la Ciudad de Guatemala
                map = L.map('map').setView([14.634915, -90.506882], 12); // Coordenadas iniciales y nivel de zoom
                L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);

                // Agregar un marcador inicial en las coordenadas iniciales
                marker = L.marker([14.634915, -90.506882]).addTo(map);

                // Define un evento clic en el mapa para cambiar la ubicación del marcador y actualizar la dirección
                map.on('click', onMapClick);
            }

            // Función para manejar el evento clic en el mapa
            function onMapClick(e) {
                // Elimina el marcador existente (si lo hay)
                if (marker) {
                    map.removeLayer(marker);
                }

                // Crea un nuevo marcador en la ubicación del clic
                marker = L.marker(e.latlng).addTo(map);

                // Actualiza el campo de texto de dirección con la dirección obtenida
                // ...
                fetch(`https://nominatim.openstreetmap.org/reverse?format=json&lat=${e.latlng.lat}&lon=${e.latlng.lng}`)
                        .then(response => response.json())
                        .then(data => {
                            const address = data.display_name;
                            direccionInput.value = address; // Actualiza el valor del campo de texto
                        })
                        .catch(error => {
                            console.error('Error en la solicitud de geocodificación inversa:', error);
                            direccionInput.value = 'No se pudo obtener la dirección.';
                        });

            }
        </script>

        <!-- Incluye Leaflet y llama a initMap cuando esté listo -->
