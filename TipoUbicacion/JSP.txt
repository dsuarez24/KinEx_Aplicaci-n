        <script>
            let map;
            let marker;
            let direccionInput;

 

            function initMap() {
                direccionInput = document.getElementById('direccionInput');
                map = L.map('map').setView([14.634915, -90.506882], 12);
                L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);

 

                marker = L.marker([14.634915, -90.506882]).addTo(map);

 

                map.on('click', onMapClick);
            }

 

function onMapClick(e) {
    // Elimina el marcador existente (si lo hay)
    if (marker) {
        map.removeLayer(marker);
    }

 

    // Crea un nuevo marcador en la ubicación del clic
    marker = L.marker(e.latlng).addTo(map);

 
axios.get(`https://api.opencagedata.com/geocode/v1/json?q=${e.latlng.lat}+${e.latlng.lng}&key=1c51292f53f44d04961f19587935fe19`)
    .then(response => {
        const data = response.data;
        console.log('Respuesta de OpenCage Geocoder:', data);

 

        if (data.status && data.status.message === 'OK') {
            const components = data.results[0].components;
            let address = '';

 

            // Verificar los componentes relevantes, como la calle y la avenida
            if (components.road) {
                address += components.road;
            }

 

            if (components.avenue) {
                if (address !== '') {
                    address += ' & ';
                }
                address += components.avenue;
            }

 

            // Actualiza el valor del campo de texto
            direccionInput.value = address || 'No se pudo obtener la dirección.';
        } else {
            console.error('Error en la respuesta de OpenCage Geocoder:', data.status ? data.status.message : 'Error desconocido');
            direccionInput.value = 'No se pudo obtener la dirección.';
        }
    })
    .catch(error => {
        console.error('Error en la solicitud a OpenCage Geocoder:', error);
        direccionInput.value = 'No se pudo obtener la dirección.';
    });
}
 

            // Incluye Leaflet y llama a initMap cuando esté listo

            script.src = 'https://unpkg.com/leaflet@1.7.1/dist/leaflet.js';
            script.onload = initMap;
